                }
            };
            let currentPhraseIndex = 0;
            let currentCharIndex = 0;
            const typingSpeed = 100;
            const pauseBetweenPhrases = 2000;
            let typingInterval; 
            function typePhrase() {
                const phrasesToUse = JSON.parse(typingContainer.getAttribute(`data-${htmlTag.lang}-phrases`));
                const currentPhrase = phrasesToUse[currentPhraseIndex];
                if (currentCharIndex < currentPhrase.length) {
                    if (typingContainer.style.visibility !== 'hidden') {
                        typingContainer.textContent += currentPhrase.charAt(currentCharIndex);
                    }
                    currentCharIndex++;
                    typingInterval = setTimeout(typePhrase, typingSpeed);
                } else {
                    typingContainer.style.animation = "blink 0.75s step-end infinite";
                    setTimeout(() => {
                        typingContainer.style.visibility = 'hidden';
                        typingContainer.style.animation = "none";
                        currentPhraseIndex = (currentPhraseIndex + 1) % phrasesToUse.length;
                        currentCharIndex = 0;
                        typingContainer.textContent = ""; 
                        setTimeout(() => {
                            typingContainer.style.visibility = 'visible'; 
                            typePhrase();
                        }, 500); 
                    }, pauseBetweenPhrases);
                }
            }
            function updateContent(lang) {
                htmlTag.setAttribute('lang', lang);
                document.title = translations[lang].title;
                document.querySelector('meta[name="description"]').setAttribute('content', translations[lang].metaDescription);
                const elementsToTranslate = document.querySelectorAll('[data-es], [data-en]');
                elementsToTranslate.forEach(element => {
                    if (element.id === 'typingText') {
                        return; 
                    }
                    if (element.hasAttribute(`data-${lang}`)) {
                        element.innerHTML = element.getAttribute(`data-${lang}`);
                    }
                });
                if (hiddenEngineeringCertsDiv.classList.contains('hidden-certs')) {
                    toggleEngineeringButton.textContent = translations[lang].viewMore;
                } else {
                    toggleEngineeringButton.textContent = translations[lang].viewLess;
                }
                langToggleBtn.textContent = (lang === 'es' ? 'English' : 'Espa√±ol');
                clearTimeout(typingInterval); 
                currentPhraseIndex = 0;
                currentCharIndex = 0;
                typingContainer.textContent = ""; 
                typingContainer.style.visibility = 'visible'; 
                typePhrase();
            }
            typePhrase();
            toggleEngineeringButton.addEventListener('click', function() {
                if (hiddenEngineeringCertsDiv.classList.contains('hidden-certs')) {
                    hiddenEngineeringCertsDiv.classList.remove('hidden-certs');
                    this.textContent = translations[htmlTag.lang].viewLess;
                } else {
                    hiddenEngineeringCertsDiv.classList.add('hidden-certs');
                    this.textContent = translations[htmlTag.lang].viewMore;
                }
            });
            if (htmlTag.lang === 'es') {
                toggleEngineeringButton.textContent = translations.es.viewMore;
            } else {
                toggleEngineeringButton.textContent = translations.en.viewMore;
            }
            langToggleBtn.addEventListener('click', function() {
                let currentLang = htmlTag.getAttribute('lang');
                let newLang = (currentLang === 'es' ? 'en' : 'es');
                updateContent(newLang);
            });
            updateContent(htmlTag.getAttribute('lang'));
        });
